12#[derive(Debug, PartialEq)]
13pub enum Error {
14    LexerError(crate::lexer::Error),
15    ParserError(crate::parser::Error),
16    /// Error when trying to use an invalid prefix operator for the given type
17    InvalidPrefixOperatorForType { span: Span, operator: PrefixOperator, right: Object, operator_span: Span, right_span: Span },
18    /// Error when trying to find an identifier that does not exist
19    IdentifierNotFound { span: Span, identifier: Rc<str>, similar: Option<Rc<str>> },
20    /// Error when trying to assign to an indexed expression
21    InvalidIndexedAssignmentExpression { span: Span, left: Rc<Expression> },
22    /// Error when trying to mutate a value that is not mutable
23    MutateError { span: Span, name: Rc<str> },
24    /// Error when trying to use an unsupported infix operator for the given types (left and right)
25    UnsupportedInfixOperator {
26        left_span: Span,
27        right_span: Span,
28        left_object: Object,
29        right_object: Object,
30        operator: InfixOperator,
31        operator_span: Span
32    },
33    /// Error when trying to iterate over a non-iterable
34    IteratingOverNonIterable { 
35        obj: Object, 
36        obj_span: Span, 
37        for_span: Span,
38        reason: IntoIteratorError },
39    /// Error when trying to call a non-function
40    CannotCall { function: Object, span: Span },
41    WrongNumberOfArguments { 
42        span: Span, 
43        unexpected_arg: Option<(usize, Argument)>,
44        expected: RangeInclusive<usize>, found: usize },
45    BuiltInError { span: Span, message: String, error_type: BuiltinError },
46    IterCallOnNonIterable { 
47        obj: Object,
48        obj_span: Span, 
49        iter_call_span: Span, reason: IntoIteratorError },
50    TypeError { span: Span, expected: Vec<ObjectTy>, found: ObjectTy, context: Option<Span> },
51    AssertionFailed { span: Span, assert_span: Span, value: Object },
52    AssertionEqualFailed { assert_span: Span, first_span: Span, second_span: Span, first_value: Object, second_value: Object },
53    IOError { error: IOError, span: Option<Span> },
54    MutateNonExistentVariable { name: Rc<str> },
55    IndexOutOfBounds { array_span: Span, index_span: Span, index: usize, length: usize },
56    IteratorMethodOnIterable { method_span: Span, method_name: Rc<str>, object_type: ObjectTy, object_span: Span },
57    PropertyNotFound { span: Span, property: Rc<str>, object_type: ObjectTy },
58    IndexingNonIndexableType { indexed_span: Span, indexed_obj: Object },
59}

61#[derive(Debug)]
62pub struct IOError {

67impl PartialEq for IOError {

69        true

73impl From<std::io::Error> for IOError {
