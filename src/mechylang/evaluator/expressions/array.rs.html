1use crate::{Environment, Evaluator, Object};
2use crate::parser::expressions::{ArrayLiteral, ExpressionSpanExt, IndexExpression};
3use crate::evaluator::{Result, Error};
4use crate::evaluator::objects::ObjectTy;
5
6impl Evaluator {
7    pub(super) fn eval_array_expression(
8        &mut self,
9        array: &ArrayLiteral,
10        env: &mut Environment,
11    ) -> Result<Object> {
12        let elements = array.elements
13            .iter()
14            .map(|element| self.eval_expression(element, env))
15            .collect::<Result<Vec<Object>>>()?;
16
17        Ok(Object::Array(elements))
18    }
19
20
21    pub(super) fn eval_index_expression(
22        &mut self,
23        index: &IndexExpression,
24        env: &mut Environment,
25    ) -> Result<Object> {
26        let left = self.eval_expression(&index.left, env)?;
27        let array = left.as_array().ok_or_else(|| Error::TypeError {
28            span: index.left_span.clone(),
29            expected: vec![ObjectTy::Array { expected_item_types: None }],
30            found: left.get_type(),
31            context: None,
32        })?;
33
34        let evaluated_index = self.eval_expression(&index.index, env)?;
35        let evaluated_index = evaluated_index.as_integer().ok_or_else(|| Error::TypeError {
36            span: index.index_span.clone(),
37            expected: vec![ObjectTy::Integer],
38            found: evaluated_index.get_type(),
39            context: None,
40        })? as usize;
41        
42        match array.get(evaluated_index) {
43            Some(item) => Ok(item.clone()),
44            None => Err(Error::IndexOutOfBounds {
45                array_span: index.left.span().clone(),
46                index_span: index.index.span().clone(),
47                index: evaluated_index,
48                length: array.len(),
49            }.into()),
50        }
51    }
52}
