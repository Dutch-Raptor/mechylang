1use std::rc::Rc;
2use crate::{Environment, Evaluator, Object,  trace};
3use crate::parser::expressions::{ExpressionSpanExt, InfixExpression, InfixOperator};

6impl Evaluator {
7    pub(super) fn eval_infix_expression(
8        &mut self,
9        infix: &InfixExpression,
10        env: &mut Environment,
11    ) -> Result<Object> {
12        let _trace = trace!(&format!("Evaluating infix expression: {}", infix).to_string());
13        // check if the infix operator is an assignment operator
14        if infix.operator.is_assignment() {
15            return self.eval_assignment_expression(infix, env);
16        }
17
18        let left = self.eval_expression(&infix.left, env)?;
19        let right = self.eval_expression(&infix.right, env)?;
20
21        let left_span = infix.left.span().clone();
22        let right_span = infix.right.span().clone();
23        let operator_span = infix.operator_span.clone();
24
25        self.current_span = infix.span.clone();
26
27        match (&left, &right) {
28            (Object::Unit, _) | (_, Object::Unit) => {
29                self.eval_unit_infix_expression(&infix.operator, &left, &right)
30            }
31            (Object::Integer(left), Object::Integer(right)) => {
32                self.eval_integer_infix_expression(&infix.operator, *left, *right)
33            }
34            (Object::Float(left), Object::Float(right)) => {
35                self.eval_float_infix_expression(&infix.operator, *left, *right)
36            }
37            (Object::Float(left), Object::Integer(right)) => {
38                self.eval_float_infix_expression(&infix.operator, *left, *right as f64)
39            }
40            (Object::Integer(left), Object::Float(right)) => {
41                self.eval_float_infix_expression(&infix.operator, *left as f64, *right)
42            }
43            (Object::Boolean(left), Object::Boolean(right)) => {
44                self.eval_boolean_infix_expression(&infix.operator, *left, *right)
45            }
46            (Object::String(left), Object::String(right)) => {
47                self.eval_string_infix_expression(&infix.operator, left.clone(), right.clone())
48            }
49            (Object::Array(left), Object::Array(right)) => {
50                self.eval_array_infix_expression(&infix.operator, left.clone(), right.clone())
51            }
52            (left, right) if infix.operator == InfixOperator::CompareEqual => {
53                Some(Object::Boolean(left == right))
54            }
55            (left, right) if infix.operator == InfixOperator::CompareNotEqual => {
56                Some(Object::Boolean(left != right))
57            }
58            _ => None
59        }.ok_or_else(|| Error::UnsupportedInfixOperator {
60            left_span,
61            right_span,
62            left_object: left.clone(),
63            right_object: right.clone(),
64            operator: infix.operator,
65            operator_span,
66        }.into())
67    }

70    fn eval_unit_infix_expression(
71        &self,
72        operator: &InfixOperator,
73        left: &Object,
74        right: &Object,
75    ) -> Option<Object> {
76        match operator {
77            InfixOperator::CompareNotEqual => Some(Object::Boolean(left != right)),
78            InfixOperator::CompareEqual => Some(Object::Boolean(left == right)),
79            _ => None,
80        }
81    }

83    fn eval_integer_infix_expression(
84        &self,
85        operator: &InfixOperator,
86        left: i64,
87        right: i64,
88    ) -> Option<Object> {
89        match operator {
90            InfixOperator::Plus => Some((left + right).into()),
91            InfixOperator::Minus => Some((left - right).into()),
92            InfixOperator::Asterisk => Some((left * right).into()),
93            InfixOperator::Slash => Some((left / right).into()),
94            InfixOperator::Percent => Some((left % right).into()),
95
96            InfixOperator::CompareEqual => Some((left == right).into()),
97            InfixOperator::CompareNotEqual => Some((left != right).into()),
98            InfixOperator::CompareGreater => Some((left > right).into()),
99            InfixOperator::CompareLess => Some((left < right).into()),
100            InfixOperator::CompareGreaterEqual => Some((left >= right).into()),
101            InfixOperator::CompareLessEqual => Some((left <= right).into()),
102
103            InfixOperator::BitwiseOr => Some((left | right).into()),
104            InfixOperator::BitwiseAnd => Some((left & right).into()),
105            InfixOperator::BitwiseXor => Some((left ^ right).into()),
106            InfixOperator::BitwiseLeftShift => Some((left << right).into()),
107            InfixOperator::BitwiseRightShift => Some((left >> right).into()),
108
109            // Explicitly not supported. This ensures that we always handle all possible operators
110            InfixOperator::LogicalAnd | InfixOperator::LogicalOr | InfixOperator::AssignEqual
111            | InfixOperator::AssignPlus | InfixOperator::AssignMinus | InfixOperator::AssignAsterisk
112            | InfixOperator::AssignSlash | InfixOperator::AssignPercent | InfixOperator::AssignBitwiseOr
113            | InfixOperator::AssignBitwiseAnd | InfixOperator::AssignBitwiseXor
114            => None,
115        }
116    }

129            InfixOperator::Slash => Some((left / right).into()),
130            InfixOperator::Percent => Some((left % right).into()),
131            InfixOperator::CompareEqual => Some((left == right).into()),
132            InfixOperator::CompareNotEqual => Some((left != right).into()),
133            InfixOperator::CompareGreater => Some((left > right).into()),
134            InfixOperator::CompareLess => Some((left < right).into()),
135            InfixOperator::CompareGreaterEqual => Some((left >= right).into()),
136            InfixOperator::CompareLessEqual => Some((left <= right).into()),
137
138            // Explicitly not supported. This ensures that we always handle all possible operators
139            InfixOperator::LogicalAnd | InfixOperator::LogicalOr | InfixOperator::BitwiseOr
140            | InfixOperator::BitwiseAnd | InfixOperator::BitwiseXor | InfixOperator::BitwiseLeftShift
141            | InfixOperator::BitwiseRightShift | InfixOperator::AssignEqual | InfixOperator::AssignPlus
142            | InfixOperator::AssignMinus | InfixOperator::AssignAsterisk | InfixOperator::AssignSlash
143            | InfixOperator::AssignPercent | InfixOperator::AssignBitwiseOr | InfixOperator::AssignBitwiseAnd
144            | InfixOperator::AssignBitwiseXor => None,
145        }

148    fn eval_boolean_infix_expression(
149        &self,
150        operator: &InfixOperator,
151        left: bool,
152        right: bool,
153    ) -> Option<Object> {
154        match operator {
155            InfixOperator::CompareEqual => Some((left == right).into()),
156            InfixOperator::CompareNotEqual => Some((left != right).into()),
157            InfixOperator::LogicalAnd => Some((left && right).into()),
158            InfixOperator::LogicalOr => Some((left || right).into()),
159
160            // Explicitly not supported. This ensures that we always handle all possible operators
161            InfixOperator::Plus | InfixOperator::Minus | InfixOperator::Asterisk
162            | InfixOperator::Slash | InfixOperator::Percent | InfixOperator::CompareGreater
163            | InfixOperator::CompareLess | InfixOperator::CompareGreaterEqual | InfixOperator::CompareLessEqual
164            | InfixOperator::BitwiseAnd | InfixOperator::BitwiseOr | InfixOperator::BitwiseXor
165            | InfixOperator::BitwiseLeftShift | InfixOperator::BitwiseRightShift | InfixOperator::AssignEqual
166            | InfixOperator::AssignPlus | InfixOperator::AssignMinus | InfixOperator::AssignAsterisk
167            | InfixOperator::AssignSlash | InfixOperator::AssignPercent | InfixOperator::AssignBitwiseOr
168            | InfixOperator::AssignBitwiseAnd | InfixOperator::AssignBitwiseXor
169            => None,
170        }
171    }

173    fn eval_string_infix_expression(
174        &self,
175        operator: &InfixOperator,
176        left: Rc<str>,
177        right: Rc<str>,
178    ) -> Option<Object> {
179        let _trace = trace!(&format!(
180            "eval_string_infix_expression({:?}, {}, {})",
181            operator, left, right
182        ));
183        match operator {
184            InfixOperator::Plus => Some((left.to_string() + &right).into()),
185            InfixOperator::CompareEqual => Some((left == right).into()),
186            InfixOperator::CompareNotEqual => Some((left != right).into()),
187
188            // Explicitly not supported. This ensures that we always handle all possible operators
189            InfixOperator::Minus | InfixOperator::Asterisk | InfixOperator::Slash
190            | InfixOperator::Percent | InfixOperator::CompareGreater | InfixOperator::CompareLess
191            | InfixOperator::CompareGreaterEqual | InfixOperator::CompareLessEqual | InfixOperator::LogicalAnd
192            | InfixOperator::LogicalOr | InfixOperator::BitwiseOr | InfixOperator::BitwiseAnd
193            | InfixOperator::BitwiseXor | InfixOperator::BitwiseLeftShift | InfixOperator::BitwiseRightShift
194            | InfixOperator::AssignEqual | InfixOperator::AssignPlus | InfixOperator::AssignMinus
195            | InfixOperator::AssignAsterisk | InfixOperator::AssignSlash | InfixOperator::AssignPercent
196            | InfixOperator::AssignBitwiseOr | InfixOperator::AssignBitwiseAnd | InfixOperator::AssignBitwiseXor
197            => None,
198        }
199    }

201    fn eval_array_infix_expression(&self, operator: &InfixOperator, left: Vec<Object>, right: Vec<Object>) -> Option<Object> {
202        match operator {
203            InfixOperator::Plus => Some(Object::Array(left.into_iter().chain(right).collect())),
204            InfixOperator::CompareEqual => Some(Object::Boolean(left == right)),
205            InfixOperator::CompareNotEqual => Some(Object::Boolean(left != right)),
206
207            // Explicitly not supported. This ensures that we always handle all possible operators
208
209            InfixOperator::CompareGreater | InfixOperator::CompareLess | InfixOperator::CompareGreaterEqual 
210            | InfixOperator::CompareLessEqual | InfixOperator::LogicalAnd | InfixOperator::LogicalOr 
211            | InfixOperator::BitwiseOr | InfixOperator::BitwiseAnd | InfixOperator::BitwiseXor 
212            | InfixOperator::BitwiseLeftShift | InfixOperator::BitwiseRightShift | InfixOperator::AssignEqual 
213            | InfixOperator::AssignPlus | InfixOperator::AssignMinus | InfixOperator::AssignAsterisk | InfixOperator::AssignSlash
214            | InfixOperator::AssignPercent | InfixOperator::AssignBitwiseOr | InfixOperator::AssignBitwiseAnd
215            | InfixOperator::Minus | InfixOperator::Asterisk | InfixOperator::Slash | InfixOperator::Percent
216            | InfixOperator::AssignBitwiseXor => None,
217        }
218    }
