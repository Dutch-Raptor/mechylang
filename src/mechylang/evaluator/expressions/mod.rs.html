18impl Evaluator {
19    pub(super) fn eval_expression(
20        &mut self,
21        expression: &Expression,
22        env: &mut Environment,
23    ) -> Result<Object> {
24        let _trace = trace!(&format!("Evaluating expression: {}", expression).to_string());
25        match expression {
26            Expression::IntegerLiteral(lit) => Ok(Object::Integer(lit.value)),
27            Expression::FloatLiteral(lit) => Ok(Object::Float(lit.value)),
28            Expression::Boolean(boolean) => Ok(boolean.value.into()),
29            Expression::Prefix(prefix) => self.eval_prefix_expression(env, prefix),
30            Expression::Infix(infix) => self.eval_infix_expression(infix, env),
31            Expression::If(if_expr) => self.eval_if_expression(if_expr, env),
32            Expression::Block(block) => self.eval_scoped_block_expression(block, env),
33            Expression::Identifier(ident) => self.eval_identifier(ident, env),
34            Expression::Function(func) => Ok(Object::Function(Function {
35                params: func.parameters.clone(),
36                span: func.span.clone(),
37                body: func.body.clone(),
38                env: env.clone(),
39            })),
40            Expression::Call(call) => self.eval_call_expression(call, env),
41            Expression::StringLiteral(lit) => Ok(Object::String(lit.value.clone())),
42            Expression::Unit(_) => Ok(Object::Unit),
43            Expression::ArrayLiteral(array) => self.eval_array_expression(array, env),
44            Expression::Index(index) => self.eval_index_expression(index, env),
45            Expression::For(for_expr) => self.eval_for_expression(for_expr, env),
46            Expression::While(while_expr) => self.eval_while_expression(while_expr, env),
47            Expression::Range(_)
48            | Expression::RangeFrom(_)
49            | Expression::RangeTo(_)
50            | Expression::RangeFull(_) => self.eval_range_expression(expression, env),
51            Expression::Member(member) => self.eval_member_expression(member, env),
52            Expression::StructLiteral(lit) => self.eval_struct_expression(lit, env),
53        }
54    }
55
56    pub(super) fn is_truthy(condition: &Object) -> bool {
57        match condition {
58            Object::Unit => false,
59            Object::Boolean(boolean) => *boolean,
60            _ => true,
61        }
62    }
