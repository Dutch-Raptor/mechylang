1use std::sync::Arc;
2use lazy_static::lazy_static;
3use crate::evaluator::Error;
4use crate::evaluator::objects::{ArgumentList, FunctionTy, MethodTy, ObjectTy};
5use crate::Object;
6
7use super::MethodInner;

9lazy_static! {
10    /// ## `not() -> Boolean`
11    /// Returns the opposite of the boolean
12    ///
13    /// ```rust
14    /// # mechylang::test_utils::test_eval_ok(r#"
15    /// assert_eq(true.not(), false);
16    /// assert_eq(false.not(), true);
17    /// # "#);
18    /// ```
19    pub static ref BOOLEAN_METHODS: Arc<[MethodInner]> = Arc::new([MethodInner {
20        method_ty: MethodTy {
21            method_name: "not",
22            self_ty: Box::new(ObjectTy::Boolean),
23            function_ty: FunctionTy {
24                arguments: ArgumentList::None,
25                expected_return_type: Some(Box::new(ObjectTy::Boolean)),
26            },
27        },
28        function: |args| {
29            let boolean = args.obj.as_boolean().ok_or_else(|| Error::TypeError {
30                span: args.method_span.clone(),
31                expected: vec![ObjectTy::Boolean],
32                found: args.obj.get_type(),
33                context: Some(args.call_span.clone()),
34            })?;
35
36            Ok(Object::Boolean(!boolean))
37        },
38    }]);
39}
