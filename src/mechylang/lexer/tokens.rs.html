6#[derive(Debug, PartialEq, Clone, Default, Serialize)]
7pub struct Span {
8    /// The range of bytes in the source file
9    pub bytes: Range<usize>,
10    pub file: Option<Arc<str>>,
11}
12
13impl Span {
14    pub fn length(&self) -> usize {
15        self.bytes.end - self.bytes.start
16    }
17    
18    pub fn start(&self) -> usize {
19        self.bytes.start
20    }
21    
22    pub fn end(&self) -> usize {
23        self.bytes.end
24    }
25    
26    pub fn bytes(&self) -> Range<usize> {
27        self.bytes.clone()
28    }
29}

31#[derive(Debug, PartialEq, Clone, Serialize)]
32pub struct Token {

340
341        match self {

346            While => write!(f, "while"),
347            For => write!(f, "for"),

361            // Literals

424            EOF => write!(f, "EOF"),
425            Ellipsis => write!(f, "Ellipsis"),
