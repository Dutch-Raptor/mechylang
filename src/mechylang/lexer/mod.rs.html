17/// Implement the Lexer struct
18///
19/// The Lexer struct is responsible for lexing the input string into tokens.
20impl Lexer<'_> {
21    /// Create a new Lexer
22    ///
23    /// # Arguments
24    /// * `input` - A String slice that holds the input to be lexed
25    ///
26    /// # Example
27    /// ```
28    /// use mechylang::Lexer;
29    ///
30    /// let input = String::from(
31    ///     r#"
32    ///     fn main() {
33    ///     print(\"Hello, World!\");
34    ///     }"#
35    /// );
36    /// let lexer = Lexer::new(&input);
37    /// ```
38    pub fn new(input: &str) -> Lexer {
39        Lexer {
40            whole: input,
41            rest: input,
42            byte: 0,
43            file: None,
44        }
45    }
46
47    /// Keeps advancing the lexer until it finds a non-whitespace character
48    fn skip_whitespace(&mut self) {
49        let first_non_whitespace = self.rest.find(|c: char| !c.is_whitespace()).unwrap_or(self.rest.len());
50        self.rest = &self.rest[first_non_whitespace..];
51        self.byte += first_non_whitespace;
52    }
53
54    /// Read an identifier from the current position. An identifier is a sequence of letters, digits and underscores.
55    fn read_identifier(&mut self) -> String {
56        let first_non_ident = self.rest.find(|c: char| !(c.is_alphabetic() || c == '_' || c.is_numeric()))
57            .unwrap_or(self.rest.len());
58        let start_byte = self.byte;
59        let end_byte = self.byte + first_non_ident;
60
61        // update positions
62        self.byte += first_non_ident;
63        self.rest = &self.rest[first_non_ident..];
64
65        self.whole[start_byte..end_byte].to_string()
66    }
67
68    /// Read a number from the input string.
69    ///
70    /// This can be an integer or a floating point number.
71    ///
72    /// Underscores are allowed in numbers, but they are ignored.
73    fn read_number(&mut self) -> String {
74        const PERIOD_LEN_BYTES: usize = '.'.len_utf8();
75        let start_byte = self.byte;
76
77        let first_non_digit = self.rest
78            .find(|c| !matches!(c, '0'..='9' | '_'))
79            .unwrap_or(self.rest.len());
80
81        self.byte += first_non_digit;
82        self.rest = &self.rest[first_non_digit..];
83
84        // If the next character is a dot, and the second next character is a digit,
85        // then we have a floating point number.
86        let c = self.rest.chars().next().unwrap_or('\0');
87        if c == '.' && self.peek_char().is_ascii_digit() {
88            // Here we skip the first char in self.rest as it as a `.`
89            let second_non_digit = self.rest[PERIOD_LEN_BYTES..]
90                .find(|c| !matches!(c, '0'..='9' | '_'))
91                .map(|at| at + PERIOD_LEN_BYTES) // add the extra byte for `.`
92                .unwrap_or(self.rest.len());
93
94            self.byte += second_non_digit;
95            self.rest = &self.rest[second_non_digit..];
96        }
97
98        self.whole[start_byte..self.byte].replace("_", "")
99    }
100
101    /// Peeks the 2nd char in `self.rest`, if None, reads '\0'
102    fn peek_char(&mut self) -> char {
103        self.rest.chars().nth(1).unwrap_or('\0')
104    }
105
106
107    /// Peeks the 3rd char in `self.rest`, if None, reads '\0'
108    fn peek_second_char(&mut self) -> char {
109        self.rest.chars().nth(2).unwrap_or('\0')
110    }

113    /// Advances the lexer by the length of `token_kind` and returns `token_kind`
114    ///
115    /// # Panics:
116    /// - if `token_kind` is a keyword, identifier, number or string
117    fn read_token(&mut self, token_kind: TokenKind) -> TokenKind {
118        let len = token_kind.length_bytes().expect("Expected token to have a const known length");
119        self.byte += len;
120        self.rest = &self.rest[len..];
121        token_kind
122    }
123
124    /// Get the next token from the input string.
125    ///
126    /// This will skip whitespace and return the next token.
127    /// If the end of the input string is reached, next_token will return None
128    pub fn next_token(&mut self) -> Option<Result<Token>> {
129        self.skip_whitespace();
130
131        if self.rest.is_empty() {
132            return None;
133        }
134
135        let c = self.rest.chars().next()?;
136        let c_start = self.byte;
137
138
139        let token_kind = match c {
140            '=' => {
141                if self.peek_char() == '=' {
142                    self.read_token(TokenKind::CompareEqual)
143                } else {
144                    self.read_token(TokenKind::AssignEqual)
145                }
146            }
147            ':' => self.read_token(TokenKind::Colon),
148            ';' => self.read_token(TokenKind::Semicolon),
149            '(' => self.read_token(TokenKind::LeftParen),
150            ')' => self.read_token(TokenKind::RightParen),
151            ',' => self.read_token(TokenKind::Comma),
152            '+' => {
153                if self.peek_char() == '=' {
154                    self.read_token(TokenKind::AssignPlus)
155                } else {
156                    self.read_token(TokenKind::Plus)
157                }
158            }
159            '-' => {
160                if self.peek_char() == '=' {
161                    self.read_token(TokenKind::AssignMinus)
162                } else {
163                    self.read_token(TokenKind::Minus)
164                }
165            }
166            '!' => {
167                if self.peek_char() == '=' {
168                    self.read_token(TokenKind::CompareNotEqual)
169                } else {
170                    self.read_token(TokenKind::Bang)
171                }
172            }
173            '~' => self.read_token(TokenKind::BitwiseNot),
174            '/' => {
175                // Support comments
176                if self.peek_char() == '/' {
177                    let line_end = self.rest.find('\n').unwrap_or(self.rest.len());
178                    self.byte += line_end;
179                    self.rest = &self.rest[line_end..];
180                    return self.next_token();
181                    // Support multiline comments
182                } else if self.peek_char() == '*' {
183                    let comment_end = self.rest.find("*/")
184                        // Add 2 for the `*` and `/`
185                        .map(|i| i + 2)
186                        .unwrap_or(self.rest.len());
187                    self.byte += comment_end;
188                    self.rest = &self.rest[comment_end..];
189                    return self.next_token();
190                } else if self.peek_char() == '=' {
191                    self.read_token(TokenKind::AssignDivide)
192                } else {
193                    self.read_token(TokenKind::Divide)
194                }
195            }
196            '*' => {
197                if self.peek_char() == '=' {
198                    self.read_token(TokenKind::AssignMultiply)
199                } else {
200                    self.read_token(TokenKind::Multiply)
201                }
202            }
203            '<' => {
204                if self.peek_char() == '=' {
205                    self.read_token(TokenKind::CompareLessEqual)
206                } else if self.peek_char() == '<' {
207                    self.read_token(TokenKind::BitwiseLeftShift)
208                } else {
209                    self.read_token(TokenKind::CompareLess)
210                }
211            }
212            '%' => {
213                if self.peek_char() == '=' {
214                    self.read_token(TokenKind::AssignModulo)
215                } else {
216                    self.read_token(TokenKind::Modulo)
217                }
218            }
219            '>' => {
220                if self.peek_char() == '=' {
221                    self.read_token(TokenKind::CompareGreaterEqual)
222                } else if self.peek_char() == '>' {
223                    self.read_token(TokenKind::BitwiseRightShift)
224                } else {
225                    self.read_token(TokenKind::CompareGreater)
226                }
227            }
228            '&' => {
229                if self.peek_char() == '&' {
230                    self.read_token(TokenKind::LogicalAnd)
231                } else if self.peek_char() == '=' {
232                    self.read_token(TokenKind::AssignBitwiseAnd)
233                } else {
234                    self.read_token(TokenKind::Ampersand)
235                }
236            }
237            '|' => {
238                if self.peek_char() == '|' {
239                    self.read_token(TokenKind::LogicalOr)
240                } else if self.peek_char() == '=' {
241                    self.read_token(TokenKind::AssignBitwiseOr)
242                } else {
243                    self.read_token(TokenKind::BitwiseOr)
244                }
245            }
246            '^' => {
247                if self.peek_char() == '=' {
248                    self.read_token(TokenKind::AssignBitwiseXor)
249                } else {
250                    self.read_token(TokenKind::BitwiseXor)
251                }
252            }
253            '.' => {
254                if self.peek_char() == '.' && self.peek_second_char() == '.' {
255                    self.read_token(TokenKind::Ellipsis)
256                } else if self.peek_char() == '.' && self.peek_second_char() == '=' {
257                    self.read_token(TokenKind::RangeInclusive)
258                } else if self.peek_char() == '.' {
259                    self.read_token(TokenKind::RangeExclusive)
260                } else {
261                    self.read_token(TokenKind::Dot)
262                }
263            }
264            '{' => self.read_token(TokenKind::LeftSquirly),
265            '}' => self.read_token(TokenKind::RightSquirly),
266            '[' => self.read_token(TokenKind::LeftSquare),
267            ']' => self.read_token(TokenKind::RightSquare),
268            '\0' => return None,
269            '\"' => {
270                let literal = match self.read_string() {
271                    Ok(literal) => literal,
272                    Err(err) => {
273                        return Some(Err(err));
274                    }
275                };
276                TokenKind::String(literal)
277            }
278            c if c.is_alphabetic() || c == '_' => {
279                let literal = self.read_identifier();
280                if let Some(keyword_token) = TokenKind::is_keyword(&literal) {
281                    keyword_token
282                } else {
283                    TokenKind::Identifier(literal)
284                }
285            }
286            c if c.is_ascii_digit() => {
287                let literal = self.read_number();
288                TokenKind::Number(literal)
289            }
290            c => {
291                self.byte += c.len_utf8();
292                self.rest = &self.rest[c.len_utf8()..];
293                return Some(Err(Error::IllegalCharacter {
294                    span: Span {
295                        bytes: c_start..c_start + c.len_utf8(),
296                        file: self.file.clone(),
297                    },
298                    found: c,
299                }));
300            }
301        };
302
303        let token = Token {
304            kind: token_kind,
305            span: Span {
306                bytes: c_start..self.byte,
307                file: self.file.clone(),
308            },
309        };
310
311        Some(Ok(token))
312    }

314    fn read_string(&mut self) -> Result<String> {
315        const QUOTE_CHAR_LEN: usize = '"'.len_utf8();
316        let string_start_byte = self.byte;
317        let literal_start_byte = self.byte + QUOTE_CHAR_LEN;
318
319        // Read the opening quote
320        self.byte += QUOTE_CHAR_LEN;
321        self.rest = &self.rest[QUOTE_CHAR_LEN..];
322
323        // allow for escaped quotes
324        let literal_length_bytes = {
325            let mut chars = self.rest.char_indices();
326
327            loop {
328                let (c_at, c) = chars.next().ok_or_else(||
329                    Error::UnterminatedString {
330                        span: Span {
331                            bytes: string_start_byte..self.whole.len(),
332                            file: self.file.clone(),
333                        }
334                    })?;
335                if c == '"' {
336                    break c_at;
337                }
338                if c == '\\' {
339                    // Make sure we don't count an escaped `"` as the end
340                    chars.next();
341                }
342            }
343        };
344
345        let literal_end_byte = literal_start_byte + literal_length_bytes;
346
347        // read rest of string
348        self.byte += literal_length_bytes + QUOTE_CHAR_LEN;
349        self.rest = &self.rest[(literal_length_bytes + QUOTE_CHAR_LEN)..];
350
351        let escaped_literal = {
352            let mut escaped_literal = String::with_capacity(literal_length_bytes);
353            let mut literal = self.whole[literal_start_byte..literal_end_byte].char_indices();
354
355            while let Some((c_at, c)) = literal.next() {
356                if c != '\\' {
357                    escaped_literal.push(c);
358                    continue;
359                }
360
361                let (c2_at, c2) = literal.next().expect("literal should not end with a `\\`");
362                match c2 {
363                    '\\' => escaped_literal.push('\\'),
364                    '"' => escaped_literal.push('"'),
365                    'n' => escaped_literal.push('\n'),
366                    'r' => escaped_literal.push('\r'),
367                    't' => escaped_literal.push('\t'),
368                    _ => return Err(Error::UnsupportedEscapeSequence {
369                        span: Span {
370                            file: self.file.clone(),
371                            bytes: (literal_start_byte + c_at)..(literal_start_byte + c2_at + c2.len_utf8()),
372                        }
373                    })
374                }
375            }
376            escaped_literal
377        };
378
379        Ok(escaped_literal)
380    }

400    fn lex_token_kinds(input: &str) -> Result<Vec<TokenKind>> {
