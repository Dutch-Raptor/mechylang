21/// Represents the various types of statements in Mechylang.
22///
23/// A statement in Mechylang can be one of the following:
24#[derive(Debug, PartialEq, Serialize)]
25pub enum Statement {
26    /// A `let` statement for variable declarations.
27    ///
28    /// Syntax:
29    /// ```
30    /// # use mechylang::test_utils::test_parse_ok;
31    /// # test_parse_ok(r#"
32    /// let x = 5;
33    /// # "#)
34    /// ```
35    Let(LetStatement),
36    /// A `return` statement for returning values from functions.
37    ///
38    /// Syntax:
39    /// ```
40    /// # use mechylang::test_utils::test_parse_ok;
41    /// # test_parse_ok(r#"
42    /// fn add(a, b) {
43    ///     return a + b;
44    /// }
45    /// # "#)
46    /// ```
47    Return(ReturnStatement),
48    /// An expression statement for general expressions.
49    ///
50    /// Syntax:
51    /// ```
52    /// # use mechylang::test_utils::test_parse_ok;
53    /// # test_parse_ok(r#"
54    /// 5 + 10;
55    /// # "#)
56    /// ```
57    Expression(ExpressionStatement),
58    /// A `break` statement for breaking out of loops.
59    ///
60    /// Syntax:
61    /// ```
62    /// # use mechylang::test_utils::test_parse_ok;
63    /// # test_parse_ok(r#"
64    /// let x = 0;
65    /// while (x < 10) {
66    ///     if (x == 5) {
67    ///         break;
68    ///     }
69    ///     x = x + 1;
70    /// }
71    /// # "#)
72    /// ```
73    Break(BreakStatement),
74    /// A `continue` statement for continuing to the next iteration of loops.
75    ///
76    /// Syntax:
77    /// ```
78    /// # use mechylang::test_utils::test_parse_ok;
79    /// # test_parse_ok(r#"
80    /// let x = 0;
81    /// let sum = 0;
82    /// while (x < 10) {
83    ///     x = x + 1;
84    ///     if (x % 2 == 0) {
85    ///         continue;
86    ///     }
87    ///     sum = sum + x;
88    /// }
89    /// # "#)
90    /// ```
91    Continue(ContinueStatement),
92    /// A function declaration statement.
93    ///
94    /// Syntax:
95    /// ```
96    /// # use mechylang::test_utils::test_parse_ok;
97    /// # test_parse_ok(r#"
98    /// fn add(a, b) {
99    ///     return a + b;
100    /// }
101    /// # "#)
102    /// ```
103    Function(FunctionStatement),
104}
105impl Display for Statement {
106    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
107        match self {
108            Statement::Let(s) => write!(f, "{}", s),
109            Statement::Return(s) => write!(f, "{}", s),
110            Statement::Expression(s) => write!(f, "{}", s),
111            Statement::Break(s) => write!(f, "{}", s),
112            Statement::Continue(s) => write!(f, "{}", s),
113            Statement::Function(s) => write!(f, "{}", s),
114        }
115    }
116}

118impl Parser<'_> {
119    /// Parses a single statement in Mechylang.
120    ///
121    /// This function handles the parsing of various types of statements, including `let`, `return`,
122    /// `break`, `continue`, and function statements. If the current token does not match any of these
123    /// specific types, it treats the statement as an expression statement.
124    ///
125    /// # Returns
126    ///
127    /// Returns a `Result` containing either:
128    /// - `Ok(Statement)` if the statement was successfully parsed.
129    /// - `Err(Error)` if there was an error during parsing, such as an unexpected token.
130    ///
131    /// # Errors
132    ///
133    /// This function returns an error if:
134    /// - An error occurs while parsing an expression or any other statement type.
135    /// - After parsing a statement, the next token is not a statement terminator.
136    pub(crate) fn parse_statement(&mut self) -> Result<Statement> {
137        let _trace = trace!("parse_statement");
138
139        let statement = match self.cur_token.kind {
140            TokenKind::Let => Statement::Let(self.parse_let_statement()?),
141            TokenKind::Return => Statement::Return(self.parse_return_statement()?),
142            TokenKind::Semicolon => {
143                self.next_token()?;
144                self.parse_statement()?
145            }
146            TokenKind::Break => Statement::Break(self.parse_break_statement()?),
147            TokenKind::Continue => Statement::Continue(ContinueStatement { span: self.cur_token.span.clone() }),
148            TokenKind::Fn => self.parse_function_statement()?,
149            _ => Statement::Expression(self.parse_expression_statement()?),
150        };
151
152
153        if !Parser::is_statement_terminator(&self.peek_token) {
154            // Skip unterminated statement error for now
155            // return Err(Error::UnterminatedStatement { span: self.cur_token.span.clone() });
156        }
157
158        if self.peek_token.kind == TokenKind::Semicolon {
159            self.next_token()?;
160        }
161        Ok(statement)
162    }
163
164
165    /// Determines if the current token is a statement terminator in Mechylang.
166    ///
167    /// This function checks if the current token indicates the end of a statement. The statement
168    /// terminators include semicolons (`;`), closing braces (`}`), end of file (`EOF`), closing
169    /// parentheses (`)`), closing square brackets (`]`), and the `else` keyword. Additionally, if the
170    /// current token is on a different line than the previous token, it is considered a statement
171    /// terminator.
172    ///
173    /// # Arguments
174    ///
175    /// * `current_token` - The current token.
176    /// * `previous_token` - The token that was parsed just before the current token.
177    ///
178    /// # Returns
179    ///
180    /// * `true` if the current token is a statement terminator.
181    /// * `false` otherwise. 
182    fn is_statement_terminator(current_token: &Token) -> bool {
183        if current_token.kind == TokenKind::Semicolon
184            || current_token.kind == TokenKind::RightSquirly
185            || current_token.kind == TokenKind::EOF
186            || current_token.kind == TokenKind::RightParen
187            || current_token.kind == TokenKind::RightSquare
188            || current_token.kind == TokenKind::Else
189        {
190            return true;
191        }
192
193        false
194    }
