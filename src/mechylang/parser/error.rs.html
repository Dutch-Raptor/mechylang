3pub type Result<T> = std::result::Result<T, Error>;

5#[derive(Debug, PartialEq, Clone)]
6pub enum Error {
7    /// Lexer error
8    LexerError(crate::lexer::Error),
9    /// Expected a statement to end, but found `{found}`
10    UnterminatedStatement { span: Span },
11    /// Expected one of the following tokens, but found `{found}`
12    UnexpectedToken { span: Span, expected: Vec<TokenKind>, found: TokenKind, location: Option<Location> },
13    /// Expected a prefix operator, but found `{found}`
14    InvalidPrefix { span: Span, found: TokenKind },
15    /// Expected a number, but found `{found}`
16    InvalidNumber { span: Span, found: TokenKind },
17    /// Expected a struct key, but found `{found}`
18    InvalidStructKey { span: Span, found: TokenKind },
19    /// Expected an infix operator, but found `{found}`
20    InvalidInfix { span: Span, found: TokenKind },
21    /// Expected an expression list to either contain an(other) expression or the end token, but found `{found}`
22    UnexpectedExpressionListEnd { list_span: Span, expected_end_token: TokenKind, found: TokenKind, parse_expression_error: Option<Box<Error>> },
23    IncompleteReturnStatemen { return_span: Span, found: TokenKind },
24    MissingPrecedence { span: Span, for_token_kind: TokenKind },
25}
26
27impl std::fmt::Display for Error {
28    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
29        write!(f, "{self:?}")
30    }
31}

50            Location::FunctionStatement => write!(f, "function statement"),
51            Location::LetStatement => write!(f, "let statement"),
