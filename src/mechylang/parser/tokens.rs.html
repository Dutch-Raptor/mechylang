8impl Parser<'_> {
9    /// Advances the parser to the next token in the input stream.
10    ///
11    /// This method retrieves the next token from the lexer and updates the
12    /// `cur_token` and `peek_token` fields accordingly.
13    ///
14    /// If there are no more tokens available from the lexer, it sets the `cur_token`
15    /// to `EOF` (End of File) and adjusts its position based on the last peeked token.
16    pub(crate) fn next_token(&mut self) -> Result<()> {
17        let token = self.lexer.next_token();
18        let next = match token {
19            Some(Ok(token)) => token,
20            Some(Err(err)) => return Err(Error::LexerError(err)),
21            None => {
22                let peek_token_end_byte = self.peek_token.span.bytes.end;
23                Token {
24                    kind: TokenKind::EOF,
25                    span: Span {
26                        bytes: peek_token_end_byte..peek_token_end_byte + 1,
27                        file: self.peek_token.span.file.clone(),
28                    },
29                }
30            },
31        };
32
33        self.cur_token = std::mem::replace(&mut self.peek_token, next);
34        Ok(())
35    }

37    /// Checks if the current token matches the expected token kind.
38    ///
39    /// If the current token matches the expected `token`, returns `Ok(())`.
40    /// If the current token does not match the expected `token`, returns an `Error`.
41    ///
42    /// # Arguments
43    ///
44    /// * `token` - The expected `TokenKind` to compare against the current token.
45    ///
46    /// # Returns
47    ///
48    /// Returns `Ok(())` if the current token matches the expected `token`.
49    /// Returns an `Error` otherwise, indicating an unexpected token.
50    pub(crate) fn expect_current(&self, token: TokenKind, location: Option<Location>) -> Result<()> {
51        if self.cur_token.kind == token {
52            Ok(())
53        } else {
54            Err(
55                Error::UnexpectedToken {
56                    span: self.cur_token.span.clone(),
57                    expected: vec![token],
58                    found: self.cur_token.kind.clone(),
59                    location,
60                }
61            )
62        }
63    }
64
65    /// Checks if the next token (peek token) matches the expected token kind and advances to the next token.
66    ///
67    /// If the next token (peek token) matches the expected `token`, advances to consume it and returns `Ok(())`.
68    /// If the next token (peek token) does not match the expected `token`, returns an `Error`.
69    ///
70    /// # Arguments
71    ///
72    /// * `token` - The expected `TokenKind` to compare against the next token (peek token).
73    ///
74    /// # Returns
75    ///
76    /// Returns `Ok(())` if the next token (peek token) matches the expected `token` and advances to consume it.
77    /// Returns an `Error` otherwise, indicating an unexpected token.
78    pub(crate) fn expect_peek(&mut self, expected_token_kind: TokenKind, location: Option<Location>) -> Result<()> {
79        if self.peek_token.kind == expected_token_kind {
80            self.next_token()?;
81            Ok(())
82        } else {
83            Err(
84                Error::UnexpectedToken {
85                    span: self.peek_token.span.clone(),
86                    expected: vec![expected_token_kind],
87                    found: self.peek_token.kind.clone(),
88                    location,
89                }
90            )
91        }
92    }

94    /// Checks if the current token matches the expected token kind.
95    ///
96    /// This method compares the `kind` of the current token (`cur_token`) with the provided `token`
97    /// and returns `true` if they match, or `false` otherwise.
98    ///
99    /// # Parameters
100    /// - `token`: The `TokenKind` to compare against the `cur_token`.
101    ///
102    /// # Returns
103    /// - `bool`: `true` if the `cur_token`'s kind matches the provided `token`; otherwise, `false`.
104    pub(crate) fn is_cur_token(&self, token: TokenKind) -> bool {
105        self.cur_token.kind == token
106    }
107    /// Checks if the peek token matches the expected token kind.
108    ///
109    /// This method compares the `kind` of the peek token (`peek_token`) with the provided `token`
110    /// and returns `true` if they match, or `false` otherwise.
111    ///
112    /// # Parameters
113    /// - `token`: The `TokenKind` to compare against the `peek_token`.
114    ///
115    /// # Returns
116    /// - `bool`: `true` if the `peek_token`'s kind matches the provided `token`; otherwise, `false`.
117    pub(crate) fn is_peek_token(&self, token: TokenKind) -> bool {
118        self.peek_token.kind == token
119    }
120
121    /// Creates a new `Span` with the given start position. And the end position is the current token's end position.
122    ///
123    /// # Arguments
124    /// * `start` - The start position of the new `Span`.
125    ///
126    /// # Returns
127    /// A new `Span` with the given start position and the end position set to the current token's end position.
128    pub(super) fn span_with_start(&self, start: &Span) -> Span {
129        Span {
130            bytes: start.bytes.start..self.cur_token.span.bytes.end,
131            file: self.cur_token.span.file.clone(),
132        }
133    }
