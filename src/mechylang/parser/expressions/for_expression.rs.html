14#[derive(Debug, PartialEq, Clone, Serialize)]
15pub struct ForExpression {
16    pub span: Span,
17    pub iterator: Identifier,
18    pub iterable: Rc<Expression>,
19    pub body: BlockExpression,
20    pub index: Option<Identifier>,
21    pub else_block: Option<BlockExpression>,
22}
23
24impl Display for ForExpression {
25    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
26        write!(
27            f,
28            "for {} in {} {}",
29            match &self.index {
30                Some(index) => format!("({}, {})", index, self.iterator),
31                None => format!("{}", self.iterator),
32            },
33            self.iterable,
34            self.body
35        )
36    }
37}
38
39impl Parser<'_> {
40
41    /// Parses a for expression
42    ///
43    /// valid syntax:
44    /// ```text
45    /// for <iterator> in <iterable> { <body> }
46    /// for (<index>, <iterator>) in <iterable> { <body> }
47    /// ```
48    pub(super) fn parse_for_expression(&mut self) -> Result<ForExpression> {
49        let _trace = trace!("parse_for_expression");
50        let start = self.cur_token.span.clone();
51
52        self.next_token()?;
53
54        // check if we have a for expression with an index
55        let index = match self.cur_token.kind {
56            TokenKind::LeftParen => {
57                self.next_token()?;
58                let index_ident = self.parse_identifier()?;
59                
60                self.expect_peek(TokenKind::Comma, Some(Location::Expression))?;
61                self.next_token()?;
62                Some(index_ident)
63            }
64            _ => None,
65        };
66
67        let iterator = self.parse_identifier()?;
68
69        if index.is_some() {
70            self.expect_peek(TokenKind::RightParen, Some(Location::Expression))?;
71        }
72
73        self.expect_peek(TokenKind::In, Some(Location::Expression))?;
74        self.next_token()?;
75        
76
77        let iterable = self.parse_expression(Precedence::Lowest)?;
78
79        // set current token to the '{' token
80        self.expect_peek(TokenKind::LeftSquirly, Some(Location::Expression))?;
81
82        let body = self.parse_block_expression()?;
83
84        let else_block = self.parse_else_block()?;
85
86        Ok(ForExpression {
87            span: self.span_with_start(&start),
88            iterator,
89            iterable: Rc::new(iterable),
90            body,
91            index,
92            else_block,
93        })
94    }
95}

98#[cfg(test)]
99mod tests {
100    use crate::parser::expressions::Expression;
101    use crate::parser::tests::parse;
102    use crate::parser::statements::Statement;
103
104    #[test]
105    fn test_for_loop() {
106        let input = "for i in 1..5 { i; }";
107
108        let statements = parse(input).unwrap();
109
110        assert_eq!(statements.len(), 1);
111
112        match &statements[0] {
113            Statement::Expression(ref expr) => match expr.expression {
114                Expression::For(ref for_loop) => {
115                    assert_eq!(for_loop.iterator.to_string(), "i");
116                    assert_eq!(for_loop.iterable.to_string(), "(1..5)");
117                    assert_eq!(for_loop.body.statements.len(), 1);
118                }
119                _ => panic!("expected for loop expression"),
120            },
121            _ => panic!("expected expression statement"),
122        };
123    }

125    #[test]
126    fn test_for_loop_with_index() {
127        let input = "for (i, j) in 1..5 { i; j; }";
128
129        let statements = parse(input).unwrap();
130
131        assert_eq!(statements.len(), 1);
132
133        match &statements[0] {
134            Statement::Expression(ref expr) => match expr.expression {
135                Expression::For(ref for_loop) => {
136                    println!("{:#?}", for_loop);
137                    assert_eq!(for_loop.iterator.to_string(), "j");
138                    assert_eq!(for_loop.iterable.to_string(), "(1..5)");
139                    assert_eq!(for_loop.body.statements.len(), 2);
140
141                    match for_loop.index {
142                        Some(ref index) => {
143                            assert_eq!(index.to_string(), "i");
144                        }
145                        None => panic!("expected index"),
146                    }
147                }
148                _ => panic!("expected for loop expression"),
149            },
150            _ => panic!("expected expression statement"),
151        };
