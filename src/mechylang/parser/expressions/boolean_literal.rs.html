1use std::fmt;
2use std::fmt::{Display, Formatter};
3use serde::Serialize;
4use crate::parser::{Parser, Error, Result};
5use crate::{Span, TokenKind, trace};
6use crate::parser::error::Location;
7
8#[derive(Debug, PartialEq, Clone, Serialize)]
9pub struct BooleanLiteral {
10    pub span: Span,
11    pub value: bool,
12}
13
14impl Display for BooleanLiteral {
15    fn fmt(&self, f: &mut Formatter) -> fmt::Result {
16        write!(f, "{}", self.value)
17    }
18}
19
20impl Parser<'_> {
21    pub(super) fn parse_boolean(&self) -> Result<BooleanLiteral> {
22        let _trace = trace!("parse_boolean");
23        let value = match self.cur_token.kind {
24            TokenKind::True => true,
25            TokenKind::False => false,
26            _ => {
27                return Err(Error::UnexpectedToken {
28                    span: self.cur_token.span.clone(),
29                    expected: vec![TokenKind::True, TokenKind::False],
30                    found: self.cur_token.kind.clone(),
31                    location: Some(Location::Expression),
32                })
33            }
34        };
35
36        Ok(BooleanLiteral { span: self.cur_token.span.clone(), value })
37    }
38}
