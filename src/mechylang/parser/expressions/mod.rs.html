1//! # Parsing Expressions
2//!
3//! This module contains the parsing logic for expressions.
4//!
5//! ## Expressions
6//!
7//! An expression is a combination of values, variables, operators, and functions that are interpreted to produce another value.
8//! For example, `1 + 2` is an expression that evaluates to `3`.
9//! Expressions can be as simple as a single value, or as complex as a function call with multiple arguments.
10//!
11//!
12//! This file contains the following:
13//! - `Expression` enum
14//! - All expression variants
15//! - implementations for `Display` for all expression variants
16//! - implementations for the parse functions for all expression variants for the parser

62/// Represents an expression in Mechylang.
63///
64/// An `Expression` can be a variety of constructs such as literals, identifiers, operations, or
65/// more complex expressions like function calls and conditionals. This enum defines all the
66/// possible types of expressions that can be encountered while parsing Mechylang code.
67///
68/// # Variants
69///
70/// * `Identifier(Identifier)` - Represents a variable or function name.
71/// * `IntegerLiteral(IntegerLiteral)` - Represents an integer literal.
72/// * `FloatLiteral(FloatLiteral)` - Represents a floating-point literal.
73/// * `Prefix(PrefixExpression)` - Represents a prefix operation (e.g., `!`, `-`).
74/// * `Infix(InfixExpression)` - Represents an infix operation (e.g., `+`, `-`, `*`, `/`).
75/// * `Boolean(BooleanLiteral)` - Represents a boolean literal (`true` or `false`).
76/// * `If(IfExpression)` - Represents an if-else conditional expression.
77/// * `Function(FunctionLiteral)` - Represents a function literal (lambda or named function).
78/// * `Call(CallExpression)` - Represents a function call expression.
79/// * `Block(BlockExpression)` - Represents a block of statements.
80/// * `StringLiteral(StringLiteral)` - Represents a string literal.
81/// * `ArrayLiteral(ArrayLiteral)` - Represents an array literal.
82/// * `Index(IndexExpression)` - Represents an indexing operation (e.g., accessing an array element).
83/// * `Range(RangeExpression)` - Represents a range expression (e.g., `1..10`).
84/// * `RangeTo(RangeToExpression)` - Represents a range-to expression (e.g., `..10`).
85/// * `RangeFrom(RangeFromExpression)` - Represents a range-from expression (e.g., `1..`).
86/// * `RangeFull(RangeFullExpression)` - Represents a full range expression (e.g., `..`).
87/// * `For(ForExpression)` - Represents a for loop expression.
88/// * `While(WhileExpression)` - Represents a while loop expression.
89/// * `Member(MemberExpression)` - Represents a member access expression (e.g., accessing a field of a struct).
90/// * `Unit(Token)` - Represents a unit value, often used for expressions that evaluate to no value (`()`).
91/// * `StructLiteral(StructLiteral)` - Represents a struct literal (instantiation of a struct).

298            Expression::Boolean(boolean) => &boolean.span,
299            Expression::If(if_expr) => &if_expr.span,
300            Expression::Function(func) => &func.span,
301            Expression::Call(call) => &call.span,
302            Expression::Block(block) => &block.span,
303            Expression::StringLiteral(lit) => &lit.span,
304            Expression::ArrayLiteral(lit) => &lit.span,
305            Expression::Index(index) => &index.span,
306            Expression::Range(range) => &range.span,
307            Expression::RangeTo(range) => &range.span,
308            Expression::RangeFrom(range) => &range.span,
309            Expression::RangeFull(range) => &range.span,
310            Expression::For(for_expr) => &for_expr.span,
311            Expression::While(while_expr) => &while_expr.span,
312            Expression::Member(member) => &member.span,
313            Expression::Unit(span) => span,
314        }
315    }

347impl Parser<'_> {
348    /// Parses an expression in Mechylang with the given precedence.
349    ///
350    /// This method parses an expression from the current token, handling different levels of precedence. It supports
351    /// prefix and infix operators, and processes the expression according to the provided precedence level. The function
352    /// starts by parsing a prefix expression, then continues to parse infix expressions based on the precedence of operators.
353    ///
354    /// # Arguments
355    ///
356    /// * `precedence` - The precedence level to determine the order of operations for infix expressions. Higher precedence
357    ///   values mean that the operator binds more tightly.
358    ///
359    /// # Returns
360    ///
361    /// Returns a `Result` which is:
362    /// * `Ok(Expression)` containing the parsed expression if successful. The expression may include prefix and infix
363    ///   operations, and is structured according to the precedence rules.
364    /// * `Err(Error)` if there was an error during parsing. This includes cases where the current token does not have a
365    ///   valid prefix operator or where an unexpected token is encountered.
366    ///
367    /// # Errors
368    ///
369    /// This function returns an error if:
370    /// * The current token does not have a valid prefix operator, which indicates a parsing issue or invalid expression.
371    /// * An infix operator is expected but not found, or the precedence of operators is not respected.
372    ///
373    /// # Notes
374    ///
375    /// The `parse_expression` function assumes that the expression parsing is initiated with a valid prefix operator
376    /// and then proceeds to handle infix operators based on their precedence. The function uses the provided precedence
377    /// level to correctly parse and associate operations, ensuring that expressions are parsed according to Mechylang's
378    /// operator precedence rules.
379    pub(crate) fn parse_expression(&mut self, precedence: Precedence) -> Result<Expression> {
380        let _trace = trace!("parse_expression");
381
382        let mut left_exp = self.parse_prefix()?;
383
384        while !self.is_peek_token(TokenKind::Semicolon) && precedence < self.peek_precedence() {
385            let peek_token_kind = self.peek_token.kind.clone();
386
387            if !self.has_infix(&peek_token_kind) {
388                return Ok(left_exp);
389            }
390
391            self.next_token()?;
392
393            left_exp = self.parse_infix(left_exp)?;
394        }
395
396        Ok(left_exp)
397    }

400    pub(super) fn parse_grouped_expression(&mut self) -> Result<Expression> {
401        let _trace = trace!("parse_grouped_expression");
402        debug_assert!(self.is_cur_token(TokenKind::LeftParen), "Expected current token to be `(`");
403        let start = self.cur_token.span.clone();
404        self.next_token()?;
405
406        if self.cur_token.kind == TokenKind::RightParen {
407            return Ok(Expression::Unit(
408                self.span_with_start(&start)
409            ));
410        }
411
412        let expression = self.parse_expression(Precedence::Lowest)?;
413
414        self.expect_peek(TokenKind::RightParen, Some(Location::Expression))?;
415
416        Ok(expression)
417    }
418
419    pub(super) fn parse_expression_list(&mut self, end: TokenKind) -> Result<Vec<Expression>> {
420        let mut arguments = Vec::new();
421
422        let start_span = self.cur_token.span.clone();
423
424        if self.peek_token.kind == end {
425            self.next_token()?;
426            return Ok(arguments);
427        }
428
429        self.next_token()?;
430
431        let expression = match self.parse_expression(Precedence::Lowest) {
432            Err(Error::InvalidPrefix { found, span }) =>
433                return Err(Error::UnexpectedExpressionListEnd {
434                    list_span: self.cur_token.span.clone(),
435                    expected_end_token: end.clone(),
436                    found: found.clone(),
437                    parse_expression_error: Some(Box::new(Error::InvalidPrefix { span, found })),
438                }),
439            Err(e) => return Err(e),
440            Ok(e) => e,
441        };
442
443        arguments.push(expression);
444
445        while self.peek_token.kind == TokenKind::Comma {
446            self.next_token()?;
447
448            // allow trailing commas
449            if self.peek_token.kind == end {
450                break;
451            }
452
453            self.next_token()?;
454            arguments.push(self.parse_expression(Precedence::Lowest)?);
455        }
456
457        if self.peek_token.kind != end {
458            return Err(Error::UnexpectedExpressionListEnd {
459                list_span: self.span_with_start(&start_span),
460                expected_end_token: end.clone(),
461                found: self.peek_token.kind.clone(),
462                parse_expression_error: None,
463            });
464        }
465
466        self.next_token()?;
467
468        Ok(arguments)
469    }

477    #[test]
478    fn test_parse_expression_with_prefix_and_infix() {
479        let source_code = r#"
480        -x + y * z
481        "#;
482        let mut parser = Parser::from_source(source_code);
483        // read tokens into cur and peek
484        parser.next_token().unwrap();
485        parser.next_token().unwrap();
486
487        let result = parser.parse_expression(Precedence::Lowest);
488        assert!(result.is_ok());
489
490        if let Ok(Expression::Infix(InfixExpression { left, operator, right, .. })) = result {
491            // Check the infix expression
492            if let Expression::Prefix(
493                PrefixExpression {
494                    right: ref expr,
495                    operator: PrefixOperator::Minus,
496                    ..
497                }
498            ) = *left {
499                assert!(matches!(expr.as_ref(), Expression::Identifier(_)));
500            } else {
501                panic!("Expected prefix expression for left operand");
502            }
503
504            assert_eq!(operator, InfixOperator::Plus);
505
506            if let Expression::Infix(
507                InfixExpression {
508                    left: ref right_left,
509                    operator: ref right_op,
510                    right: ref right_right,
511                    ..
512                }
513            ) = *right {
514                assert!(matches!(right_left.as_ref(), Expression::Identifier(_)));
515                assert_eq!(right_op, &InfixOperator::Asterisk);
516                assert!(matches!(right_right.as_ref(), Expression::Identifier(_)));
517            } else {
518                panic!("Expected infix expression for right operand");
519            }
520        } else {
521            panic!("Expected infix expression");
522        }
523    }
