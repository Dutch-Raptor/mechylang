59impl InfixOperator {
60    pub fn is_assignment(&self) -> bool {
61        matches!(self, InfixOperator::AssignEqual
62        | InfixOperator::AssignPlus | InfixOperator::AssignMinus
63        | InfixOperator::AssignAsterisk | InfixOperator::AssignSlash
64        | InfixOperator::AssignPercent | InfixOperator::AssignBitwiseAnd
65        | InfixOperator::AssignBitwiseOr | InfixOperator::AssignBitwiseXor)
66    }

68    /// Returns the related operator to an assignment operator.
69    /// For example, the `AssignPlus` operator is related to the `Plus` operator.
70    /// 
71    /// For non-assignment operators, returns `None`.
72    pub fn assignment_related_operator(&self) -> Option<InfixOperator> {
73        match self {
74            InfixOperator::AssignPlus => Some(InfixOperator::Plus),
75            InfixOperator::AssignMinus => Some(InfixOperator::Minus),
76            InfixOperator::AssignAsterisk => Some(InfixOperator::Asterisk),
77            InfixOperator::AssignSlash => Some(InfixOperator::Slash),
78            InfixOperator::AssignPercent => Some(InfixOperator::Percent),
79            InfixOperator::AssignBitwiseAnd => Some(InfixOperator::BitwiseAnd),
80            InfixOperator::AssignBitwiseOr => Some(InfixOperator::BitwiseOr),
81            InfixOperator::AssignBitwiseXor => Some(InfixOperator::BitwiseXor),
82            _ => None,
83        }
84    }

122impl Parser<'_> {
123    pub(super) fn has_infix(&self, token: &TokenKind) -> bool {
124        match token {
125            TokenKind::Plus | TokenKind::Minus | TokenKind::Divide
126            | TokenKind::Multiply | TokenKind::Modulo | TokenKind::CompareEqual
127            | TokenKind::CompareNotEqual | TokenKind::CompareGreater | TokenKind::CompareGreaterEqual
128            | TokenKind::CompareLess | TokenKind::CompareLessEqual
129
130            // Open parenthesis for function calls
131            | TokenKind::LeftParen
132            // Open square bracket for array indexing
133            | TokenKind::LeftSquare
134
135            // logical operators
136            | TokenKind::LogicalAnd | TokenKind::LogicalOr
137
138            // bitwise operators
139            | TokenKind::Ampersand | TokenKind::BitwiseOr | TokenKind::BitwiseXor
140            | TokenKind::BitwiseLeftShift | TokenKind::BitwiseRightShift
141
142            // Assignment operators
143            | TokenKind::AssignEqual | TokenKind::AssignPlus | TokenKind::AssignMinus
144            | TokenKind::AssignMultiply | TokenKind::AssignDivide | TokenKind::AssignModulo
145            | TokenKind::AssignBitwiseAnd | TokenKind::AssignBitwiseOr | TokenKind::AssignBitwiseXor
146
147            // Range operator
148            | TokenKind::RangeExclusive | TokenKind::RangeInclusive
149
150            // Method expressions
151            | TokenKind::Dot
152
153            => true,
154            _ => false,
155        }
156    }
157
158    /// Parses an infix expression in Mechylang.
159    ///
160    /// This function takes the left-hand side of an expression and the current token,
161    /// and parses it into a corresponding infix `Expression`. It handles various infix
162    /// operators such as arithmetic, comparison, logical, bitwise, and assignment operators,
163    /// as well as function calls, indexing, member access, and range expressions.
164    ///
165    /// # Parameters
166    ///
167    /// - `left`: The left-hand side `Expression` of the infix operation.
168    ///
169    /// # Returns
170    ///
171    /// Returns a `Result<Expression, Error>` where:
172    /// - `Ok(Expression)` is returned if the current token is successfully parsed into an `Expression`.
173    /// - `Err(Error)` is returned if the current token does not correspond to a valid infix operator
174    ///   or if parsing fails.
175    ///
176    /// # Errors
177    ///
178    /// The function will return an error if:
179    /// - The `TokenKind` does not have a registered infix function. This could occur if the
180    ///   token type is unexpected or unsupported, resulting in an `ErrorKind::MissingInfix`
181    ///   error with a descriptive message.
182    pub(super) fn parse_infix(&mut self, left: Expression) -> Result<Expression> {
183        let _trace = trace!("parse_infix");
184        match self.cur_token.kind {
185            // **** infix operators ****
186            // Arithmetic operators
187            TokenKind::Plus | TokenKind::Minus | TokenKind::Divide
188            | TokenKind::Multiply | TokenKind::Modulo
189
190            // Comparison operators
191            | TokenKind::CompareEqual | TokenKind::CompareNotEqual | TokenKind::CompareGreater
192            | TokenKind::CompareGreaterEqual | TokenKind::CompareLess | TokenKind::CompareLessEqual
193
194            // Logical operators
195            | TokenKind::LogicalAnd | TokenKind::LogicalOr
196
197            // Bitwise operators
198            | TokenKind::Ampersand | TokenKind::BitwiseOr | TokenKind::BitwiseXor
199            | TokenKind::BitwiseLeftShift | TokenKind::BitwiseRightShift
200
201            // Assignments
202            | TokenKind::AssignEqual | TokenKind::AssignPlus | TokenKind::AssignMinus
203            | TokenKind::AssignMultiply | TokenKind::AssignDivide | TokenKind::AssignModulo
204            | TokenKind::AssignBitwiseAnd | TokenKind::AssignBitwiseOr | TokenKind::AssignBitwiseXor
205
206            => Ok(Expression::Infix(self.parse_infix_expression(left)?)),
207
208            TokenKind::LeftParen => Ok(Expression::Call(self.parse_call_expression(left)?)),
209            TokenKind::LeftSquare => Ok(Expression::Index(self.parse_index_expression(left)?)),
210            TokenKind::Dot => Ok(Expression::Member(self.parse_member(left)?)),
211            TokenKind::RangeExclusive | TokenKind::RangeInclusive => self.parse_range_infix_expression(left),
212
213            _ => Err(Error::InvalidInfix {
214                span: self.cur_token.span.clone(),
215                found: self.cur_token.kind.clone(),
216            }),
217        }
218    }
219
220
221    pub(super) fn parse_infix_expression(&mut self, left: Expression) -> Result<InfixExpression> {
222        let _trace = trace!("parse_infix_expression");
223
224        let operator = Self::parse_infix_operator(&self.cur_token)
225            .ok_or(Error::InvalidInfix {
226                span: self.cur_token.span.clone(),
227                found: self.cur_token.kind.clone(),
228            })?;
229        
230        let operator_span = self.cur_token.span.clone();
231
232        let precedence = self.cur_precedence();
233        self.next_token()?;
234        let right = self.parse_expression(precedence)?;
235
236        Ok(InfixExpression {
237            span: self.span_with_start(&left.span().clone()),
238            operator,
239            operator_span,
240            left: Rc::new(left),
241            right: Rc::new(right),
242        })
243    }
244
245    fn parse_infix_operator(token: &Token) -> Option<InfixOperator> {
246        Some(match token.kind {
247            // Arithmetic operators
248            TokenKind::Plus => InfixOperator::Plus,
249            TokenKind::Minus => InfixOperator::Minus,
250            TokenKind::Multiply => InfixOperator::Asterisk,
251            TokenKind::Divide => InfixOperator::Slash,
252            TokenKind::Modulo => InfixOperator::Percent,
253
254            // Comparison operators
255            TokenKind::CompareEqual => InfixOperator::CompareEqual,
256            TokenKind::CompareNotEqual => InfixOperator::CompareNotEqual,
257            TokenKind::CompareLess => InfixOperator::CompareLess,
258            TokenKind::CompareGreater => InfixOperator::CompareGreater,
259            TokenKind::CompareLessEqual => InfixOperator::CompareLessEqual,
260            TokenKind::CompareGreaterEqual => InfixOperator::CompareGreaterEqual,
261
262            // Bitwise operators
263            TokenKind::BitwiseXor => InfixOperator::BitwiseXor,
264            TokenKind::Ampersand => InfixOperator::BitwiseAnd,
265            TokenKind::BitwiseOr => InfixOperator::BitwiseOr,
266            TokenKind::BitwiseLeftShift => InfixOperator::BitwiseLeftShift,
267            TokenKind::BitwiseRightShift => InfixOperator::BitwiseRightShift,
268
269            // Logical operators
270            TokenKind::LogicalAnd => InfixOperator::LogicalAnd,
271            TokenKind::LogicalOr => InfixOperator::LogicalOr,
272
273            // Assignment
274            TokenKind::AssignEqual => InfixOperator::AssignEqual,
275            TokenKind::AssignPlus => InfixOperator::AssignPlus,
276            TokenKind::AssignMinus => InfixOperator::AssignMinus,
277            TokenKind::AssignMultiply => InfixOperator::AssignAsterisk,
278            TokenKind::AssignDivide => InfixOperator::AssignSlash,
279            TokenKind::AssignModulo => InfixOperator::AssignPercent,
280            TokenKind::AssignBitwiseXor => InfixOperator::AssignBitwiseXor,
281            TokenKind::AssignBitwiseAnd => InfixOperator::AssignBitwiseAnd,
282            TokenKind::AssignBitwiseOr => InfixOperator::AssignBitwiseOr,
283
284            _ => {
285                return None;
286            }
287        })
288    }
