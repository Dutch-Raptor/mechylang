1use std::fmt;
2use std::fmt::{Display, Formatter};
3use std::rc::Rc;
4use serde::Serialize;
5use crate::{Parser, Span, TokenKind};
6use crate::parser::{Error, Result};
7use crate::parser::error::Location;
8
9#[derive(Debug, PartialEq, Clone, Serialize)]
10pub struct StringLiteral {
11    pub span: Span,
12    pub value: Rc<str>,
13}
14
15impl Display for StringLiteral {
16    fn fmt(&self, f: &mut Formatter) -> fmt::Result {
17        write!(f, "{}", self.value)
18    }
19}
20
21impl Parser<'_> {
22    pub(super) fn parse_string(&mut self) -> Result<StringLiteral> {
23        let start = self.cur_token.span.clone();
24        
25        let value = self.cur_token.kind
26            .as_string()
27            .ok_or_else(|| Error::UnexpectedToken {
28                span: self.cur_token.span.clone(),
29                expected: vec![TokenKind::String(String::new())],
30                found: self.cur_token.kind.clone(),
31                location: Some(Location::Expression),
32            })?;
33
34        Ok(StringLiteral { span: self.span_with_start(&start), value: value.into() })
35    }
36}
37#[cfg(test)]
38mod tests {
39    use crate::parser::expressions::Expression;
40    use crate::parser::tests::parse;
41    use crate::parser::statements::Statement;
42
43    #[test]
44    fn test_string_literal_expression() {
45        let input = r#""hello world";"#;
46
47        let statements = parse(input).unwrap();
48
49        assert_eq!(statements.len(), 1);
50
51        let stmt = &statements[0];
52
53        match stmt {
54            Statement::Expression(ref expr) => match expr.expression {
55                Expression::StringLiteral(ref literal) => {
56                    assert_eq!(literal.value, "hello world".into());
57                }
58                _ => panic!("expected string literal expression"),
59            },
60            _ => panic!("expected expression statement"),
61        };
62    }
