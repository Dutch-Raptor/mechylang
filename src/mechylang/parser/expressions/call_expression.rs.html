1use std::fmt;
2use std::fmt::Display;
3use std::rc::Rc;
4use serde::Serialize;
5use crate::parser::expressions::{Expression, ExpressionSpanExt};
6use crate::parser::Parser;
7use crate::{Span, TokenKind};
8use crate::parser::{Result};
9
10#[derive(Debug, PartialEq, Clone, Serialize)]
11pub struct CallExpression {
12    pub span: Span,
13    pub function: Rc<Expression>,
14    pub arguments: Rc<[Expression]>,
15}
16
17impl Display for CallExpression {
18    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
19        let args = self
20            .arguments
21            .iter()
22            .map(|a| a.to_string())
23            .collect::<Vec<String>>()
24            .join(", ");
25
26        write!(f, "{}({})", self.function, args)
27    }
28}
29impl Parser<'_> {
30    pub(super) fn parse_call_expression(&mut self, left: Expression) -> Result<CallExpression> {
31        let arguments = self.parse_expression_list(TokenKind::RightParen)?;
32
33        Ok(CallExpression {
34            span: self.span_with_start(&left.span().clone()),
35            function: Rc::new(left),
36            arguments: arguments.into(),
37        })
38    }
39}

41#[cfg(test)]
42mod tests {
43    use crate::Lexer;
44    use crate::parser::expressions::Expression;
45    use crate::parser::Parser;
46    use crate::parser::statements::Statement;
47
48    #[test]
49    fn test_call_expression_parsing() {
50        let input = "add(1, 2 * 3, 4 + 5);";
51
52        let lexer = Lexer::new(input);
53        let mut parser = Parser::new(lexer);
54        let program = parser.parse().unwrap();
55
56        assert_eq!(program.statements.len(), 1);
57
58        let stmt = &program.statements[0];
59
60        match stmt {
61            Statement::Expression(ref expr) => match expr.expression {
62                Expression::Call(ref call) => {
63                    match call.function.as_ref() {
64                        Expression::Identifier(ref ident) => {
65                            assert_eq!(ident.value, "add".into());
66                        }
67                        _ => panic!("expected identifier expression"),
68                    };
69                    assert_eq!(call.arguments.len(), 3);
70                    assert_eq!(call.arguments[0].to_string(), "1");
71                    assert_eq!(call.arguments[1].to_string(), "(2 * 3)");
72                    assert_eq!(call.arguments[2].to_string(), "(4 + 5)");
73                }
74                _ => panic!("expected call expression"),
75            },
76            _ => panic!("expected expression statement"),
77        };
78    }
